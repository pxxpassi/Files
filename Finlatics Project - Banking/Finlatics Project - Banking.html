- Are there any correlations between different attributes and the likelihood of subscribing to a term deposit?

In[8]:
df['y_numeric'] = df['y'].map({'yes': 1, 'no': 0})

numerical_vars = ['age', 'balance', 'day', 'duration', 'campaign', 'pdays', 'previous']
for var in numerical_vars:
    correlation = df['y_numeric'].corr(df[var])
    print(f"Pearson correlation coefficient between 'y' and {var}: {correlation}")
